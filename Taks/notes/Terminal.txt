                                                                    Git komandalari
                                                                    
   
git init                                                                                [istenilen bir papkada lokal git reposu yaradilmasi]

git --version                                                                       [gitin komputerde olub, olmamaginin yoxlanisi]  
 
git config --global user.name                                             [global bazada user name tesdiqi]

git config --global user.email                                             [global bazada email tesdiqi]

git add .                                                                             [papkada  olan butun fayllarin lokal git bazasina yuklenmesi]

git status                                                                            [yuklenen fayllarda deyisiklik varsa, onlarin yoxlanisi] 

git commit -m "comment"                                                   [edilen deyisikliklerin izlene bilmesi ucun lokalda yadda saxlanmasi] 

git log                                                                                 [indiye qeder edilmis butun commitler siyahisi]            

git remote add origin {githubdaki reponun linki}                  [githubdaki reponu lokal repoya baglamaq]

git branch                                                                           [hal hazirdaki reponun hansi qolunda oldugunu gorsedir]

git branch -v                                                                       [hal hazirdaki oldugunuz qoldaki son commiti gorsedir]

git branch istenilen ad                                                        [lokal repoda ikinci istenilen bir adda yeni qol yaratmaq]

git branch chekout istenilen ad                                           [yaradilmis qola kecid etmek]

git branch -m basqa ad                                                       [halhazirda ki istifade olunan qolun adini deyismek]

git branch -d qolun adi                                                        [gorsedilen addaki qolu silmek]

git push                                                                               [add ve commitden sonra papkadaki reponu githuba yuklemek]

git pull                                                                                 [githubdaki reponu lokal papkaya endirmek]

git merge                                                                             [lokal ve githubdaki repolarin eyni veziyetde olmagini yoxlamaq]
 
git log --oneline --graph --decorate --all                               [commitlerin seliqeli formada verilmesi]
